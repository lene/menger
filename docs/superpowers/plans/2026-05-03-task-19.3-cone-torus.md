```cuda
//==============================================================================
// Cone Ray Intersection Shader
// Implements ray-cone intersection with base cap
//==============================================================================
// This file is included by optix_shaders.cu - do not compile separately

//==============================================================================
// Cone Intersection Program
//==============================================================================

/**
 * Ray-cone intersection.
 *
 * Cone defined by:
 * - apex: tip position
 * - base: base center position
 * - radius: base radius
 * - axis: normalized direction from apex to base
 * - height: length(base - apex)
 * - cos_a/sin_a: cosine/sine of half-angle
 *
 * Intersection types:
 * - Lateral surface: infinite-cone quadratic, clamped to [0, height] along axis
 * - Base cap: disk at base position
 */
extern "C" __global__ void __intersection__cone() {
    const unsigned int instanceId = optixGetInstanceId();
    if (instanceId >= params.num_instances) return;
    if (!params.instance_materials)        return;
    const InstanceMaterial& mat = params.instance_materials[instanceId];
    const int cone_index = mat.texture_index;
    if (cone_index < 0 || cone_index >= static_cast<int>(params.num_cones)) return;
    if (!params.cone_data) return;
    const ConeData* cone = &params.cone_data[cone_index];

    const float3 apex = make_float3(cone->apex[0], cone->apex[1], cone->apex[2]);
    const float3 base = make_float3(cone->base[0], cone->base[1], cone->base[2]);

    const float3 axis_vec = make_float3(base.x - apex.x, base.y - apex.y, base.z - apex.z);
    const float  height   = sqrtf(axis_vec.x*axis_vec.x + axis_vec.y*axis_vec.y + axis_vec.z*axis_vec.z);
    if (height < 1e-6f) return;  // degenerate cone
    const float3 axis = make_float3(axis_vec.x/height, axis_vec.y/height, axis_vec.z/height);
    const float  r    = cone->radius;
    const float  len2 = height * height + r * r;
    const float  cos_a = height / sqrtf(len2);
    const float  sin_a = r      / sqrtf(len2);
    const float  ca2   = cos_a * cos_a;

    const float3 ro = optixGetWorldRayOrigin();
    const float3 rd = optixGetWorldRayDirection();
    const float  tmin = optixGetRayTmin();
    const float  tmax = optixGetRayTmax();

    const float3 d = make_float3(ro.x - apex.x, ro.y - apex.y, ro.z - apex.z);

    const float  dv = rd.x*axis.x + rd.y*axis.y + rd.z*axis.z;
    const float  dd = rd.x*rd.x   + rd.y*rd.y   + rd.z*rd.z;
    const float  ov = d.x*axis.x  + d.y*axis.y  + d.z*axis.z;
    const float  od = d.x*rd.x    + d.y*rd.y    + d.z*rd.z;
    const float  oo = d.x*d.x     + d.y*d.y     + d.z*d.z;

    const float A = dv*dv - ca2*dd;
    const float B = 2.0f*(dv*ov - ca2*od);
    const float C = ov*ov - ca2*oo;

    float t_hit = -1.0f;
    float3 hit_normal;
    unsigned int hit_kind = 0;  // 0=lateral, 1=cap

    // Test lateral surface
    const float disc = B*B - 4.0f*A*C;
    if (disc >= 0.0f && fabsf(A) > 1e-8f) {
        const float sq = sqrtf(disc);
        float t0 = (-B - sq) / (2.0f * A);
        float t1 = (-B + sq) / (2.0f * A);
        if (t0 > t1) { float tmp = t0; t0 = t1; t1 = tmp; }

        for (int i = 0; i < 2; i++) {
            const float t = (i == 0) ? t0 : t1;
            if (t > tmin && t < tmax) {
                const float3 P    = make_float3(ro.x + t*rd.x, ro.y + t*rd.y, ro.z + t*rd.z);
                const float3 dp   = make_float3(P.x-apex.x, P.y-apex.y, P.z-apex.z);
                const float  proj = dp.x*axis.x + dp.y*axis.y + dp.z*axis.z;
                if (proj >= 0.0f && proj <= height) {
                    // Lateral normal: radial * cos_a - axis * sin_a
                    const float3 radial_unnorm = make_float3(dp.x - proj*axis.x, dp.y - proj*axis.y, dp.z - proj*axis.z);
                    const float  rlen = sqrtf(radial_unnorm.x*radial_unnorm.x + radial_unnorm.y*radial_unnorm.y + radial_unnorm.z*radial_unnorm.z);
                    const float3 radial = (rlen > 1e-8f) ?
                        make_float3(radial_unnorm.x/rlen, radial_unnorm.y/rlen, radial_unnorm.z/rlen) :
                        make_float3(1.0f, 0.0f, 0.0f);
                    const float3 N = make_float3(radial.x*cos_a - axis.x*sin_a,
                                                  radial.y*cos_a - axis.y*sin_a,
                                                  radial.z*cos_a - axis.z*sin_a);
                    if (t_hit < 0.0f || t < t_hit) {
                        t_hit = t;
                        hit_normal = N;
                        hit_kind = 0;
                    }
                    break;  // closest lateral hit found
                }
            }
        }
    }

    // Test base cap (disk at base, normal = -axis)
    const float rd_dot_axis = rd.x*axis.x + rd.y*axis.y + rd.z*axis.z;
    if (fabsf(rd_dot_axis) > RayTracingConstants::CYLINDER_CAP_PARALLEL_THRESHOLD) {
        const float3 base_to_ro = make_float3(ro.x-base.x, ro.y-base.y, ro.z-base.z);
        const float  t_cap = -(base_to_ro.x*axis.x + base_to_ro.y*axis.y + base_to_ro.z*axis.z) / rd_dot_axis;
        if (t_cap > tmin && t_cap < tmax && (t_hit < 0.0f || t_cap < t_hit)) {
            const float3 P_cap = make_float3(ro.x + t_cap*rd.x, ro.y + t_cap*rd.y, ro.z + t_cap*rd.z);
            const float3 v_cap = make_float3(P_cap.x-base.x, P_cap.y-base.y, P_cap.z-base.z);
            const float  dist2 = v_cap.x*v_cap.x + v_cap.y*v_cap.y + v_cap.z*v_cap.z;
            if (dist2 <= r*r) {
                t_hit = t_cap;
                hit_normal = make_float3(-axis.x, -axis.y, -axis.z);
                hit_kind = 1;
            }
        }
    }

    if (t_hit > 0.0f) {
        optixReportIntersection(
            t_hit,
            hit_kind,
            __float_as_uint(hit_normal.x),
            __float_as_uint(hit_normal.y),
            __float_as_uint(hit_normal.z)
        );
    }
}

    const float t = optixGetRayTmax();
    const float3 ray_origin    = optixGetWorldRayOrigin();
    const float3 ray_direction = optixGetWorldRayDirection();
    const float3 hit_point = make_float3(
        ray_origin.x + ray_direction.x * t,
        ray_origin.y + ray_direction.y * t,
        ray_origin.z + ray_direction.z * t
    );

    const float3 normal = make_float3(
        __uint_as_float(optixGetAttribute_0()),
        __uint_as_float(optixGetAttribute_1()),
        __uint_as_float(optixGetAttribute_2())
    );

    const unsigned int depth = optixGetPayload_3();

    if (params.stats) {
        atomicMax(&params.stats->max_depth_reached, depth + 1);
        atomicMin(&params.stats->min_depth_reached, depth + 1);
    }

    float4 material_color;
    float material_ior, roughness, metallic, specular, emission, film_thickness;
    getInstanceMaterialPBR(material_color, material_ior, roughness, metallic, specular, emission, film_thickness);

    if (depth == 0 && metallic > 0.0f) {
        handleMetallicOpaque(hit_point, ray_direction, normal,
                             material_color, metallic, depth, emission);
        return;
    }

    const float3 final_lighting = calculateLighting(hit_point, normal, false, true);

    const float final_r = material_color.x * final_lighting.x * RayTracingConstants::COLOR_SCALE_FACTOR;
    const float final_g = material_color.y * final_lighting.y * RayTracingConstants::COLOR_SCALE_FACTOR;
    const float final_b = material_color.z * final_lighting.z * RayTracingConstants::COLOR_SCALE_FACTOR;

    const float emissive_r = fminf(final_r + emission * material_color.x * RayTracingConstants::COLOR_BYTE_MAX, RayTracingConstants::COLOR_BYTE_MAX);
    const float emissive_g = fminf(final_g + emission * material_color.y * RayTracingConstants::COLOR_BYTE_MAX, RayTracingConstants::COLOR_BYTE_MAX);
    const float emissive_b = fminf(final_b + emission * material_color.z * RayTracingConstants::COLOR_BYTE_MAX, RayTracingConstants::COLOR_BYTE_MAX);

    optixSetPayload_0(static_cast<unsigned int>(emissive_r));
    optixSetPayload_1(static_cast<unsigned int>(emissive_g));
    optixSetPayload_2(static_cast<unsigned int>(emissive_b));
}
```

```cuda
//==============================================================================
// Torus Ray Intersection Shader
// Implements ray-torus intersection via degree-4 polynomial (Schwarze/Cardano)
//==============================================================================
// This file is included by optix_shaders.cu - do not compile separately

//==============================================================================
// Quartic Solver (Schwarze/Ferrari-Cardano method)
// Uses double precision internally for numerical stability.
// Returns number of real roots written to roots[].
//==============================================================================

__device__ static int solveCubic(double a, double b, double c, double roots[3]) {
    // Depressed cubic: t^3 + pt + q = 0 after substitution t = u - a/3
    const double p = b - a*a/3.0;
    const double q = 2.0*a*a*a/27.0 - a*b/3.0 + c;
    const double D = q*q/4.0 + p*p*p/27.0;

    if (D > 0.0) {
        const double sqrtD = sqrt(D);
        const double u = cbrt(-q/2.0 + sqrtD);
        const double v = cbrt(-q/2.0 - sqrtD);
        roots[0] = u + v - a/3.0;
        return 1;
    } else if (D < 0.0) {
        const double r = sqrt(-p*p*p/27.0);
        const double phi = acos(-q / (2.0*r));
        const double m = 2.0 * cbrt(r);
        roots[0] = m * cos(phi/3.0)          - a/3.0;
        roots[1] = m * cos((phi + 2.0*M_PI)/3.0) - a/3.0;
        roots[2] = m * cos((phi + 4.0*M_PI)/3.0) - a/3.0;
        return 3;
    } else {
        // D == 0: double root
        const double u = cbrt(-q/2.0);
        roots[0] = 2.0*u - a/3.0;
        roots[1] = -u    - a/3.0;
        return 2;
    }
}

__device__ static int solveQuartic(double a, double b, double c, double d, double e, double roots[4]) {
    // Normalize: divide by a
    const double B = b/a, C = c/a, D = d/a, E = e/a;

    // Resolvent cubic: y^3 - C*y^2 + (B*D - 4*E)*y - (B^2*E - 4*C*E + D^2) = 0
    double cubic_roots[3];
    const int ncubic = solveCubic(-C, B*D - 4.0*E, -(B*B*E - 4.0*C*E + D*D), cubic_roots);

    // Pick the real cubic root that gives the best-conditioned quadratic factorization
    double y = cubic_roots[0];
    if (ncubic > 1) {
        for (int i = 1; i < ncubic; i++) {
            if (fabs(cubic_roots[i]) > fabs(y)) y = cubic_roots[i];
        }
    }

    // Factor into two quadratics: z^2 + p1*z + q1 and z^2 + p2*z + q2
    // where p1+p2 = B, q1+q2 = C - y, p1*q2 + p2*q1 = D
    const double disc1 = B*B/4.0 - C + y;
    if (disc1 < 0.0) return 0;
    const double sqrtDisc1 = sqrt(disc1);

    const double p1 =  B/2.0 + sqrtDisc1;
    const double p2 =  B/2.0 - sqrtDisc1;
    double q1, q2;
    if (fabs(sqrtDisc1) > 1e-10) {
        q1 = y/2.0 + (B*y/2.0 - D) / (2.0 * sqrtDisc1);
        q2 = y/2.0 - (B*y/2.0 - D) / (2.0 * sqrtDisc1);
    } else {
        q1 = y/2.0 + sqrt(y*y/4.0 - E);
        q2 = y/2.0 - sqrt(y*y/4.0 - E);
    }

    int count = 0;
    // Roots of z^2 + p1*z + q1
    double d1 = p1*p1/4.0 - q1;
    if (d1 >= 0.0) {
        const double sq = sqrt(d1);
        roots[count++] = -p1/2.0 + sq;
        roots[count++] = -p1/2.0 - sq;
    }
    // Roots of z^2 + p2*z + q2
    double d2 = p2*p2/4.0 - q2;
    if (d2 >= 0.0) {
        const double sq = sqrt(d2);
        roots[count++] = -p2/2.0 + sq;
        roots[count++] = -p2/2.0 - sq;
    }
    return count;
}

//==============================================================================
// Torus Intersection Program
//==============================================================================

/**
 * Ray-torus intersection.
 * Sprint 19: Y-axis torus only.
 *
 * Torus equation (Y-axis, centered at origin, major radius R, minor radius r):
 *   (sqrt(x^2 + z^2) - R)^2 + y^2 = r^2
 * Expanding and substituting P = center + t*rd gives a degree-4 polynomial in t.
 */
extern "C" __global__ void __intersection__torus() {
    const unsigned int instanceId = optixGetInstanceId();
    if (instanceId >= params.num_instances) return;
    if (!params.instance_materials)        return;
    const InstanceMaterial& mat = params.instance_materials[instanceId];
    const int torus_index = mat.texture_index;
    if (torus_index < 0 || torus_index >= static_cast<int>(params.num_toruses)) return;
    if (!params.torus_data) return;
    const TorusData* torus = &params.torus_data[torus_index];

    const float3 center = make_float3(torus->center[0], torus->center[1], torus->center[2]);
    const double R = torus->major_radius;
    const double r = torus->minor_radius;

    const float3 ro = optixGetWorldRayOrigin();
    const float3 rd = optixGetWorldRayDirection();
    const float  tmin = optixGetRayTmin();
    const float  tmax = optixGetRayTmax();

    // Translate ray to torus-local frame (center at origin)
    const double ox = (double)(ro.x - center.x);
    const double oy = (double)(ro.y - center.y);
    const double oz = (double)(ro.z - center.z);
    const double dx = (double)rd.x;
    const double dy = (double)rd.y;
    const double dz = (double)rd.z;

    // Quartic coefficients for (|P|^2 + R^2 - r^2)^2 = 4*R^2*(x^2 + z^2)
    const double sum_d2 = dx*dx + dy*dy + dz*dz;
    const double sum_od = ox*dx + oy*dy + oz*dz;
    const double sum_o2 = ox*ox + oy*oy + oz*oz;
    const double R2 = R*R, r2 = r*r;
    const double K  = sum_o2 - R2 - r2;

    const double a = sum_d2 * sum_d2;
    const double b = 4.0 * sum_d2 * sum_od;
    const double c = 2.0 * sum_d2 * K + 4.0 * sum_od*sum_od + 4.0 * R2 * dy*dy;
    const double d = 4.0 * K * sum_od + 8.0 * R2 * oy*dy;
    const double e = K*K - 4.0 * R2 * (oy*oy - r2) + 4.0 * R2 * oy*oy - 4.0 * R2 * r2 + 4.0 * R2 * (oy*oy);

    // Recalculate e cleanly:
    // (sum_o2 - R2 - r2)^2 - 4R^2(x^2 + z^2 - r^2) = 0
    // e = K*K - 4*R2*(ox*ox + oz*oz - r2)
    const double e_clean = K*K - 4.0*R2*(ox*ox + oz*oz - r2);

    double roots[4];
    const int nroots = solveQuartic(a, b, c, d, e_clean, roots);

    float t_hit = -1.0f;
    for (int i = 0; i < nroots; i++) {
        const float t = (float)roots[i];
        if (t > tmin && t < tmax && (t_hit < 0.0f || t < t_hit)) {
            t_hit = t;
        }
    }

    if (t_hit > 0.0f) {
        // Compute normal at hit point P
        const float3 P = make_float3(ro.x + t_hit*rd.x, ro.y + t_hit*rd.y, ro.z + t_hit*rd.z);
        const float px = P.x - center.x;
        const float py = P.y - center.y;
        const float pz = P.z - center.z;

        const float denom = sqrtf(px*px + pz*pz);
        float3 normal;
        if (denom > 1e-8f) {
            const float factor = 1.0f - (float)R / denom;
            const float nx = px * factor;
            const float ny = py;
            const float nz = pz * factor;
            const float nlen = sqrtf(nx*nx + ny*ny + nz*nz);
            normal = (nlen > 1e-8f) ? make_float3(nx/nlen, ny/nlen, nz/nlen) :
                                      make_float3(0.0f, (py > 0.0f) ? 1.0f : -1.0f, 0.0f);
        } else {
            // On the inner circle — degenerate case
            normal = make_float3(0.0f, (py > 0.0f) ? 1.0f : -1.0f, 0.0f);
        }

        optixReportIntersection(
            t_hit,
            0u,
            __float_as_uint(normal.x),
            __float_as_uint(normal.y),
            __float_as_uint(normal.z)
        );
    }
}

extern "C" __global__ void __closesthit__torus() {
    const float t = optixGetRayTmax();
    const float3 ray_origin    = optixGetWorldRayOrigin();
    const float3 ray_direction = optixGetWorldRayDirection();
    const float3 hit_point = make_float3(
        ray_origin.x + ray_direction.x * t,
        ray_origin.y + ray_direction.y * t,
        ray_origin.z + ray_direction.z * t
    );

    const float3 normal = make_float3(
        __uint_as_float(optixGetAttribute_0()),
        __uint_as_float(optixGetAttribute_1()),
        __uint_as_float(optixGetAttribute_2())
    );

    const unsigned int depth = optixGetPayload_3();

    if (params.stats) {
        atomicMax(&params.stats->max_depth_reached, depth + 1);
        atomicMin(&params.stats->min_depth_reached, depth + 1);
    }

    float4 material_color;
    float material_ior, roughness, metallic, specular, emission, film_thickness;
    getInstanceMaterialPBR(material_color, material_ior, roughness, metallic, specular, emission, film_thickness);

    if (depth == 0 && metallic > 0.0f) {
        handleMetallicOpaque(hit_point, ray_direction, normal,
                             material_color, metallic, depth, emission);
        return;
    }

    const float3 final_lighting = calculateLighting(hit_point, normal, false, true);

    const float final_r = material_color.x * final_lighting.x * RayTracingConstants::COLOR_SCALE_FACTOR;
    const float final_g = material_color.y * final_lighting.y * RayTracingConstants::COLOR_SCALE_FACTOR;
    const float final_b = material_color.z * final_lighting.z * RayTracingConstants::COLOR_SCALE_FACTOR;

    const float emissive_r = fminf(final_r + emission * material_color.x * RayTracingConstants::COLOR_BYTE_MAX, RayTracingConstants::COLOR_BYTE_MAX);
    const float emissive_g = fminf(final_g + emission * material_color.y * RayTracingConstants::COLOR_BYTE_MAX, RayTracingConstants::COLOR_BYTE_MAX);
    const float emissive_b = fminf(final_b + emission * material_color.z * RayTracingConstants::COLOR_BYTE_MAX, RayTracingConstants::COLOR_BYTE_MAX);

    optixSetPayload_0(static_cast<unsigned int>(emissive_r));
    optixSetPayload_1(static_cast<unsigned int>(emissive_g));
    optixSetPayload_2(static_cast<unsigned int>(emissive_b));
}
```

```scala
package menger.engines.scene

import scala.util.Try

import menger.ObjectSpec
import menger.common.Vector
import menger.optix.OptiXRenderer

class ConeSceneBuilder extends SceneBuilder:

  override def validate(specs: List[ObjectSpec], maxInstances: Int): Either[String, Unit] =
    if specs.isEmpty then Left("Object specs list cannot be empty")
    else if specs.length > maxInstances then Left(s"Too many objects: ${specs.length} exceeds max instances limit of $maxInstances")
    else if !specs.forall(_.objectType.toLowerCase == "cone") then Left("All objects must be cones for ConeSceneBuilder")
    else Right(())

  override def buildScene(specs: List[ObjectSpec], renderer: OptiXRenderer, maxInstances: Int): Try[Unit] = Try:
    logger.debug(s"Setting up ${specs.length} cone instances")
    specs.foreach { spec =>
      val material = MaterialExtractor.extract(spec)
      val (ax, ay, az) = spec.apex.getOrElse((spec.x, spec.y + spec.size / 2f, spec.z))
      val (bx, by, bz) = spec.base.getOrElse((spec.x, spec.y - spec.size / 2f, spec.z))
      val radius = spec.majorRadius.getOrElse(spec.size / 2f)
      val apex = Vector[3](ax, ay, az)
      val base = Vector[3](bx, by, bz)
      renderer.addConeInstance(apex, base, radius, material) match
        case Some(id) => logger.debug(s"Added cone instance $id")
        case None     => logger.error(s"Failed to add cone instance")
    }

  override def isCompatible(spec1: ObjectSpec, spec2: ObjectSpec): Boolean =
    spec1.objectType.toLowerCase == "cone" && spec2.objectType.toLowerCase == "cone"

  override def calculateInstanceCount(specs: List[ObjectSpec]): Long = specs.length.toLong
```
