```scala
package menger.objects

import com.badlogic.gdx.math.Vector3
import menger.common.TriangleMeshData
import org.scalatest.flatspec.AnyFlatSpec
import org.scalatest.matchers.should.Matchers

class PolytopesSuite extends AnyFlatSpec with Matchers:

  private val Eps = 1e-4f

  private def norm(x: Float, y: Float, z: Float): Float =
    math.sqrt(x * x + y * y + z * z).toFloat

  private def allVerticesOnUnitSphere(data: TriangleMeshData): Unit =
    val coords = data.vertices
    val stride = 3
    for i <- coords.indices by stride do
      val d = norm(coords(i), coords(i + 1), coords(i + 2))
      d shouldBe (1f +- Eps)

  private def hasPositiveOutwardNormals(data: TriangleMeshData): Unit =
    val v = data.vertices
    val idx = data.indices
    for i <- idx.indices by 3 do
      val (i0, i1, i2) = (idx(i) * 3, idx(i + 1) * 3, idx(i + 2) * 3)
      val (ax, ay, az) = (v(i0), v(i0 + 1), v(i0 + 2))
      val (bx, by, bz) = (v(i1), v(i1 + 1), v(i1 + 2))
      val (cx, cy, cz) = (v(i2), v(i2 + 1), v(i2 + 2))
      // Edge vectors
      val (e1x, e1y, e1z) = (bx - ax, by - ay, bz - az)
      val (e2x, e2y, e2z) = (cx - ax, cy - ay, cz - az)
      // Cross product (face normal)
      val (nx, ny, nz) = (
        e1y * e2z - e1z * e2y,
        e1z * e2x - e1x * e2z,
        e1x * e2y - e1y * e2x
      )
      // Centroid
      val (centX, centY, centZ) = ((ax + bx + cx) / 3f, (ay + by + cy) / 3f, (az + bz + cz) / 3f)
      // Normal should point outward (positive dot with centroid direction)
      val dot = nx * centX + ny * centY + nz * centZ
      dot should be > 0f

  "Tetrahedron" should "have 4 vertices" in:
    val data = Tetrahedron(Vector3(0f, 0f, 0f), 1f).toTriangleMesh
    data.vertices.length / 3 shouldBe 4

  it should "have 4 triangles (12 indices)" in:
    val data = Tetrahedron(Vector3(0f, 0f, 0f), 1f).toTriangleMesh
    data.indices.length shouldBe 12

  it should "have all vertices on unit sphere" in:
    val data = Tetrahedron(Vector3(0f, 0f, 0f), 1f).toTriangleMesh
    allVerticesOnUnitSphere(data)

  it should "have outward face normals" in:
    val data = Tetrahedron(Vector3(0f, 0f, 0f), 1f).toTriangleMesh
    hasPositiveOutwardNormals(data)

  it should "scale vertices by scale parameter" in:
    val data = Tetrahedron(Vector3(0f, 0f, 0f), 2f).toTriangleMesh
    val coords = data.vertices
    val stride = 3
    for i <- coords.indices by stride do
      val d = norm(coords(i), coords(i + 1), coords(i + 2))
      d shouldBe (2f +- Eps)

  it should "translate vertices by center parameter" in:
    val center = Vector3(1f, 2f, 3f)
    val data = Tetrahedron(center, 1f).toTriangleMesh
    val coords = data.vertices
    val stride = 3
    var found = false
    for i <- coords.indices by stride do
      val d = norm(coords(i) - 1f, coords(i + 1) - 2f, coords(i + 2) - 3f)
      if math.abs(d - 1f) < Eps then found = true
    found shouldBe true
```

```scala
package menger.objects

import com.badlogic.gdx.graphics.GL20
import com.badlogic.gdx.graphics.g3d.Material
import com.badlogic.gdx.graphics.g3d.ModelInstance
import com.badlogic.gdx.math.Vector3
import menger.common.TriangleMeshData
import menger.common.TriangleMeshSource

case class Icosahedron(
  center: Vector3 = Vector3.Zero,
  scale: Float = 1f,
  material: Material = Builder.WHITE_MATERIAL,
  primitiveType: Int = GL20.GL_TRIANGLES
) extends Geometry(center, scale) with TriangleMeshSource:

  @SuppressWarnings(Array("org.wartremover.warts.Throw"))
  def getModel: List[ModelInstance] =
    throw UnsupportedOperationException("Icosahedron is OptiX-only; getModel not supported")

  def toTriangleMesh: TriangleMeshData =
    val cx = center.x
    val cy = center.y
    val cz = center.z

    // Unit-sphere icosahedron vertices
    // φ = (1 + √5) / 2; normalizer = 1 / √(1 + φ²)
    val rawVertices: Array[Float] = Icosahedron.scaledVertices(cx, cy, cz, scale)

    val indices: Array[Int] = Array(
       0, 11,  5,   0,  5,  1,   0,  1,  7,   0,  7, 10,   0, 10, 11,
       1,  5,  9,   5, 11,  4,  11, 10,  2,  10,  7,  6,   7,  1,  8,
       3,  9,  4,   3,  4,  2,   3,  2,  6,   3,  6,  8,   3,  8,  9,
       4,  9,  5,   2,  4, 11,   6,  2, 10,   8,  6,  7,   9,  8,  1
    )

    TriangleMeshData(rawVertices, indices)

object Icosahedron:
  val Phi: Float = ((1.0 + math.sqrt(5.0)) / 2.0).toFloat
  // Distance from origin to a vertex of the un-normalized icosahedron (0, 1, φ)
  private val Norm: Float = math.sqrt(1.0 + Phi * Phi).toFloat

  def scaledVertices(cx: Float, cy: Float, cz: Float, scale: Float): Array[Float] =
    val n = Norm
    val s = scale
    Array(
      // Group 1: permutations of (0, ±1, ±φ) in YZ plane
       0f / n * s + cx,  1f / n * s + cy,  Phi / n * s + cz,
       0f / n * s + cx, -1f / n * s + cy,  Phi / n * s + cz,
       0f / n * s + cx,  1f / n * s + cy, -Phi / n * s + cz,
       0f / n * s + cx, -1f / n * s + cy, -Phi / n * s + cz,
      // Group 2: permutations of (±φ, 0, ±1) in XZ plane
       Phi / n * s + cx,  0f / n * s + cy,  1f / n * s + cz,
      -Phi / n * s + cx,  0f / n * s + cy,  1f / n * s + cz,
       Phi / n * s + cx,  0f / n * s + cy, -1f / n * s + cz,
      -Phi / n * s + cx,  0f / n * s + cy, -1f / n * s + cz,
      // Group 3: permutations of (±1, ±φ, 0) in XY plane
       1f / n * s + cx,  Phi / n * s + cy,  0f / n * s + cz,
      -1f / n * s + cx,  Phi / n * s + cy,  0f / n * s + cz,
       1f / n * s + cx, -Phi / n * s + cy,  0f / n * s + cz,
      -1f / n * s + cx, -Phi / n * s + cy,  0f / n * s + cz
    )
```

**Note on face indices:** The index array above produces 20 CCW-wound triangles for the vertex
layout defined in `scaledVertices`. Verify the `hasPositiveOutwardNormals` test passes; if any
face fails, flip the winding of that triangle (swap the last two indices of that row).

Dodecahedron: 20 vertices, 12 pentagonal faces fan-triangulated → 36 triangles (108 indices).

Vertices (dual of icosahedron, two groups normalized to unit sphere):
- Group A: all sign combinations of `(±1, ±1, ±1)` — 8 vertices — each at distance √3, normalize to `(±1/√3, ±1/√3, ±1/√3)`
- Group B: all even permutations of `(0, ±1/φ, ±φ)` — 12 vertices — each at distance `√(1/φ² + φ²)`, normalize accordingly

The 12 pentagonal faces each have 5 vertices. Each pentagon is fan-triangulated from its first vertex: `(v0,v1,v2), (v0,v2,v3), (v0,v3,v4)` → 3 triangles per face = 36 total.

```scala
  val Phi: Float = ((1.0 + math.sqrt(5.0)) / 2.0).toFloat
  private val InvPhi: Float = 1.0f / Phi

  // Normalizers
  // Group A: (±1, ±1, ±1) — distance = √3
  private val NormA: Float = math.sqrt(3.0).toFloat
  // Group B: (0, ±1/φ, ±φ) — distance = √(1/φ² + φ²) = √((1 + φ⁴)/φ²)
  private val NormB: Float = math.sqrt(InvPhi * InvPhi + Phi * Phi).toFloat

  private def a(v: Float): Float = v / NormA
  private def b(v: Float, u: Float): (Float, Float) = (v / NormB, u / NormB)

  // 20 vertices in order (indices match Indices array below)
  // v0–v7:  Group A — (±1, ±1, ±1) normalized
  // v8–v19: Group B — even permutations of (0, ±1/φ, ±φ) normalized
  def scaledVertices(cx: Float, cy: Float, cz: Float, scale: Float): Array[Float] =
    val s = scale
    val (bip, bph) = b(InvPhi, Phi)
    val (bnp, bnh) = b(-InvPhi, Phi)
    val (bim, bph2) = b(InvPhi, -Phi)
    val (bnm, bnh2) = b(-InvPhi, -Phi)

       a( 1f) * s + cx,  a( 1f) * s + cy,  a( 1f) * s + cz,  // v0  +++
       a(-1f) * s + cx,  a( 1f) * s + cy,  a( 1f) * s + cz,  // v1  -++

       a(-1f) * s + cx,  a(-1f) * s + cy,  a( 1f) * s + cz,  // v3  --+
       a( 1f) * s + cx,  a( 1f) * s + cy,  a(-1f) * s + cz,  // v4  ++-

       a( 1f) * s + cx,  a(-1f) * s + cy,  a(-1f) * s + cz,  // v6  +--
       a(-1f) * s + cx,  a(-1f) * s + cy,  a(-1f) * s + cz,  // v7  ---

      -bph * s + cx, 0f + cy,  bip * s + cz,  // v13  -φ, 0, +1/φ
       bph * s + cx, 0f + cy, -bip * s + cz,  // v14  +φ, 0, -1/φ
      -bph * s + cx, 0f + cy, -bip * s + cz,  // v15  -φ, 0, -1/φ
      // v16..v19: (±1/φ, ±φ, 0) normalized
       bip * s + cx,  bph * s + cy, 0f + cz,  // v16  +1/φ, +φ, 0
      -bip * s + cx,  bph * s + cy, 0f + cz,  // v17  -1/φ, +φ, 0
       bip * s + cx, -bph * s + cy, 0f + cz,  // v18  +1/φ, -φ, 0
      -bip * s + cx, -bph * s + cy, 0f + cz   // v19  -1/φ, -φ, 0
    )

  // 12 pentagonal faces, each fan-triangulated from first vertex → 3 triangles each
  // Face winding: CCW from outside. Verify with hasPositiveOutwardNormals test.
  val Indices: Array[Int] = Array(
    // Top cap faces (near +Y)

     0,  8,  9,   0,  9,  2,   0,  2, 12,   // face 1: v0,v8,v9,v2,v12

     8,  1, 13,   8, 13,  3,   8,  3,  9,   // face 5: v8,v1,v13,v3,v9
```
